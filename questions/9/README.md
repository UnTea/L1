# колько существует способов задать переменную типа `slice` или `map`?

```go
var (
    a = []int{}            // []              len=0 cap=0
    b = []int{1, 2}        // [1 2]           len=2 cap=2
    c = []int{5: 123}      // [0 0 0 0 0 123] len=6 cap=6
    d = make([]int, 5, 10) // [0 0 0 0 0]     len=5 cap=10
)

var a [4]int               // [0 0 0 0]
var a [...]int{1, 2, 3, 4} // [0 0 0 0]
```
